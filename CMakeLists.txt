CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

OPTION(BUILD_BIND_CPP "Build the C++ bindings" OFF)

IF(BUILD_BIND_CPP)
	INCLUDE_DIRECTORIES("include")

    IF(BUILD_MONOLITHIC)
        SHARED_STATIC(SIEGE++ src cpp)
    ENDIF(BUILD_MONOLITHIC)
    IF(BUILD_SPLIT)
        SHARED_STATIC(SG-Audio++ src/audio cpp)
        SHARED_STATIC(SG-Core++ src/core cpp)
        SHARED_STATIC(SG-Graphics++ src/graphics cpp)
        SHARED_STATIC(SG-Input++ src/input cpp)
    ENDIF(BUILD_SPLIT)

	PROJECT(Example-Bindings-CPP)
		ADD_EXECUTABLE(${PROJECT_NAME} main.cpp)

		TARGET_LINK_LIBRARIES(${PROJECT_NAME} SIEGE SIEGE++ ${AUX_LIBRARIES})

		INSTALL(TARGETS ${PROJECT_NAME} DESTINATION bin)
ENDIF(BUILD_BIND_CPP)
